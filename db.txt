# CREATE DATABASE IF NOT EXISTS nft;
# CREATE DATABASE IF NOT EXISTS nft_test;


DO
    $$
BEGIN
    -- Создание базы данных nft, если она не существует
    IF NOT EXISTS (SELECT 1 FROM pg_database WHERE datname = 'nft') THEN
CREATE DATABASE nft;
END IF;

   -- Создание базы данных nft_test, если она не существует
IF NOT EXISTS (SELECT 1 FROM pg_database WHERE datname = 'nft_test') THEN
CREATE DATABASE nft_test;
END IF;
END
$$
